```c
include<stdio.h>
#include<unistd.h>
int main()
{
    int pid;
    pid=fork();
    if(pid<0)
    {
        printf("fork failed");
    }
    else if(pid==0)
    {
        printf("it is the child process pid is %d, and %d\n ",getpid(),pid);
    }
    else
    {
        printf("it is a parent process and pid %d, %d\n ",getpid(),pid);
    }
}
```
